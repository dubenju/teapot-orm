017package org.teapot.db;
018
019import java.io.FileInputStream;
020import java.io.IOException;
021import java.io.PrintWriter;
022import java.sql.Connection;
023import java.sql.DriverManager;
024import java.sql.SQLException;
025import java.util.Properties;
026
027/**
028 * 与数据库相关处理的类。
029 * 通过JDBC创建或释放数据库连接。
030 * @author dubenju@126.com
031 */
032public class TeapotDb {
033
034  private static volatile TeapotDb INSTANCE;
035  private String dbType = null;
036  private String dbServer = null;
037  private String dbUser = null;
038  private String dbPwd = null;
039  private String dbSchema = null;
040
041  /**
042   * 获取实例.
043   * @return 实例（单实例）。
044   */
045  public static TeapotDb getInstance() {
046    if (TeapotDb.INSTANCE == null) {
047      synchronized (TeapotDb.class) {
048        if (TeapotDb.INSTANCE == null) {
049          TeapotDb.INSTANCE = new TeapotDb();
050          TeapotDb.INSTANCE.init();
051        }
052      }
053    }
054    return TeapotDb.INSTANCE;
055  }
056
057  /**
058   * 初期化处理，从属性配置文件中获取信息并实例化驱动.
059   */
060  public void init() {
061    String path = TeapotDb.class.getResource("").getPath();
062    Properties prop = new Properties();//属性集合对象
063    // 当前目录>conf>class
064    try {
065      //属性文件流
066      FileInputStream fis = new FileInputStream(path + "../../../../conf/Teapot_ORM.properties");
067      prop.load(fis);//将属性文件流装载到Properties对象中
068      fis.close();// 关闭流
069    } catch (IOException ex) {
070      ex.printStackTrace();
071    }
072
073    String driverName = prop.getProperty("teapot.database.driver");
074    dbType = prop.getProperty("teapot.database.type");
075    dbServer = prop.getProperty("teapot.database.hostname");
076    dbUser = prop.getProperty("teapot.database.user");
077    dbPwd = prop.getProperty("teapot.database.password");
078    dbSchema = prop.getProperty("teapot.database.schema");
079
080//    System.out.println("获取属性值：teapot.database.driver=" + driverName);
081//    System.out.println("获取属性值：teapot.database.type=" + dbType);
082//    System.out.println("获取属性值：teapot.database.hostname=" + dbServer);
083//    System.out.println("获取属性值：teapot.database.user=" + dbUser);
084//    System.out.println("获取属性值：teapot.database.password=" + dbPwd);
085//    System.out.println("获取属性值：teapot.database.schema=" + dbSchema);
086
087    try {
088      Class.forName(driverName).newInstance();
089    } catch (InstantiationException e) {
090      e.printStackTrace();
091    } catch (IllegalAccessException e) {
092      e.printStackTrace();
093    } catch (ClassNotFoundException e) {
094      e.printStackTrace();
095    }
096  }
097
098  /**
099   * 获取数据库连接.
100   * @return 获取的数据库连接。
101   */
102  public TeapotDbConnection getConnection() {
103    DriverManager.setLogWriter(new PrintWriter(System.out));
104    Connection conn = null;
105    // jdbc:mysql://localhost/teapot?user=root&password=root&useUnicode=true&characterEncoding=utf8
106    // jdbc:sqlite:test.db
107    // jdbc:odbc:TEST
108    // jdbc:oracle:thin:@dbserver:1521:oracle...
109    // jdbc:log4jdbc:derby://localhost:1527//db-derby-10.2.2.0-bin/databases/MyDatabase
110    // jdbc:datadirect:oracle://server3:1521;User=test;Password=secret
111    StringBuffer buf = new StringBuffer();
112    buf.append("jdbc:");
113    buf.append(dbType);
114    if (!"sqlite".equals(dbType) && !"odbc".equals(dbType)) {
115      buf.append("://");
116      buf.append(dbServer);
117      buf.append("/");
118    } else {
119      buf.append(":");
120    }
121    buf.append(dbSchema);
122    if ("mysql".equals(dbType)) {
123      buf.append("?user=");
124      buf.append(dbUser);
125      buf.append("&password=");
126      buf.append(dbPwd);
127      buf.append("&useUnicode=true&characterEncoding=utf8");
128    }
129    try {
130      conn = DriverManager.getConnection(buf.toString());
131    } catch (SQLException e) {
132      e.printStackTrace();
133    }
134    return new TeapotDbConnection(conn);
135  }
136
137  /**
138   * 释放已经获取的数据库连接.
139   * @param dbConn 数据库连接。
140   * @throws SQLException 处理过程中的异常。
141   */
142  public void releaseConnection(Object dbConn) throws SQLException {
143    if (dbConn instanceof Connection) {
144      ((Connection) dbConn).close();
145    }
146  }
147
148  /**
149   * 构造函数.
150   */
151  private TeapotDb() {
152  }
153
154  /**
155   * 返回数据库类型.
156   * @return 数据库类型。
157   */
158  public String getDbType() {
159    return dbType;
160  }
161}
