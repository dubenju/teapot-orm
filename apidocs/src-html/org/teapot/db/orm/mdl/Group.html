001package org.teapot.db.orm.mdl;
002
003import java.util.ArrayList;
004
005public class Group implements IClause {
006  private ArrayList<String> cols = new ArrayList<String>();
007  /*
008   * GROUP BY
009   * {
010   *     column-Name [ , column-Name ]*
011   * |
012   *     ROLLUP ( column-Name [ , column-Name ]* )
013   * }
014   *
015   * MYSQL
016     *     [GROUP BY {col_name | expr | position}
017     *       [ASC | DESC], ... [WITH ROLLUP]]
018   */
019  public Group(String col) {
020    this.cols.add(col);
021  }
022
023  public Group by(String col) {
024    this.cols.add(col);
025    return this;
026  }
027
028  /**
029   * @see java.lang.Object#toString()
030   */
031  @Override
032  public String toString() {
033    StringBuffer buf = new StringBuffer();
034    int max = this.cols.size();
035    int idx = 0;
036    while (idx < max) {
037      buf.append(this.cols.get(idx));
038      idx ++;
039      if (idx < max) {
040        buf.append(",");
041      }
042    }
043    return buf.toString();
044  }
045}
