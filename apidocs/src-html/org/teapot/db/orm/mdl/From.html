001package org.teapot.db.orm.mdl;
002
003import java.util.ArrayList;
004
005public class From implements IClause {
006  private ArrayList<Table> tables = new ArrayList<Table>();
007
008  public From(String tbl) {
009    this(tbl, tbl);
010  }
011
012  public From(String tbl, String als) {
013    this.tables.add(new Table(tbl, als));
014  }
015
016  public From inner(String tbl, String als) {
017    this.tables.add(new Table(tbl, als));
018    return this;
019  }
020
021  public From left(String tbl, String als) {
022    this.tables.add(new Table(tbl, als));
023    return this;
024  }
025
026  public From right(String tbl, String als) {
027    this.tables.add(new Table(tbl, als));
028    return this;
029  }
030
031  public From full(String tbl, String als) {
032    this.tables.add(new Table(tbl, als));
033    return this;
034  }
035
036  /**
037   * @see java.lang.Object#toString()
038   */
039  @Override
040  public String toString() {
041    StringBuffer buf = new StringBuffer();
042    int max = this.tables.size();
043    int idx = 0;
044    while (idx < max) {
045      buf.append(this.tables.get(idx));
046      idx ++;
047    }
048    return buf.toString();
049  }
050}
