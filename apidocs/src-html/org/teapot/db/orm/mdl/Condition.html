017package org.teapot.db.orm.mdl;
018
019import java.util.ArrayList;
020import java.util.List;
021
022/**
023 * where clause
024 * 数据库SQL的条件类.
025 * WHERE 子句
026 * @author dubenju@126.com
027 * @since 0.0.1
028 */
029public class Condition implements IClause {
030
031  /* 01. AND, OR, NOT */
032  public static final String AND = " AND ";
033  public static final String OR  = " OR ";
034  public static final String NOT = " NOT ";
035
036  /* 02.Comparisons */
037  public static final String GREATER = " > ";
038  public static final String LESS = " < ";
039  public static final String EQUAL = " = ";
040  public static final String GREATER_EQUAL = " >= ";
041  public static final String LESS_EQUAL = " <= ";
042  public static final String NOT_EQUAL = " != ";
043  public static final String NOT_EQUAL2 = " <> ";
044
045  /* 03.IS NULL, IS NOT NULL */
046  public static final String IS_NULL = " IS NULL ";
047  public static final String IS_NOT_NULL = " IS NOT NULL ";
048
049  /* 04.LIKE */
050  public static final String LIKE = " LIKE ";
051
052  /* 05.BETWEEN介于两个值之间的数据范围BETWEEN同AND一起搭配使用. */
053  public static final String BETWEEN = " BETWEEN ";
054
055  /* 06.IN */
056  public static final String IN = " IN ";
057
058  /* 07.EXISTS */
059  public static final  String EXISTS = " EXISTS ";
060
061  /*
062   * Quantified comparison
063   * A quantified comparison is a comparison operator (<, =, >, <=, >=, <>) with ALL or ANY or
064   * SOME applied.
065   * Operates on table subqueries, which can return multiple rows but must return a single column.
066   * If ALL is used, the comparison must be true for all values returned by the table subquery.
067   * If ANY or SOME is used, the comparison must be true for at least one value of the table
068   * subquery. ANY and SOME are equivalent.
069   *
070   * WHERE normal_rate < ALL
071   * (SELECT budget/550 FROM Groups)
072   *
073   * Expression
074   *   ComparisonOperator
075   *   {
076   *     ALL |
077   *     ANY |
078   *     SOME
079   *   }
080   *   TableSubquery
081   */
082  private StringBuffer buf;
083  private List<Object> values;
084
085  /**
086   * 构造函数.
087   * @since 0.0.1
088   */
089  public Condition() {
090    this.buf = new StringBuffer();
091    this.values = new ArrayList<>();
092  }
093
094  /**
095   * 获取条件文中所有的值.
096   * @return 条件文中所有的值。
097   * @since 0.0.1
098   */
099  public List<?> getValues() {
100    return this.values;
101  }
102
103  private void makeBuffer(String cond, Object value) {
104    if (Condition.BETWEEN.equals(cond)) {
105      this.buf.append(" ? AND ? ");
106      if (value instanceof Object[]) {
107        this.values.add(((Object[]) value)[0]);
108        this.values.add(((Object[]) value)[1]);
109      }
110      if (value instanceof List<?>) {
111        this.values.add(((List<?>) value).get(0));
112        this.values.add(((List<?>) value).get(1));
113      }
114    } else if (Condition.IN.equals(cond)) {
115      // IN 条件时的处理
116      // SELECT * FROM HOGE WHERE ID IN (?,?,?)
117      if (value instanceof Object[]) {
118        this.buf.append(" ( ");
119        Object[] coll = (Object[]) value;
120        for (int idx = 0; idx < coll.length; idx ++) {
121          Object val = coll[idx];
122          this.buf.append(" ? ");
123          if ((idx + 1) < coll.length) {
124            this.buf.append(" , ");
125          }
126          this.values.add(val);
127        }
128        this.buf.append(" ) ");
129      }
130
131      if (value instanceof List<?>) {
132        this.buf.append(" (");
133        List<?> coll = (List<?>) value;
134        for (int idx = 0; idx <  coll.size(); idx ++) {
135          Object val = coll.get(idx);
136          this.buf.append(" ? ");
137          if ((idx + 1) < coll.size()) {
138            this.buf.append(" , ");
139          }
140          this.values.add(val);
141        }
142        this.buf.append(") ");
143      }
144    } else {
145      if ((!Condition.IS_NULL.equals(cond)) && (!Condition.IS_NOT_NULL.equals(cond))) {
146        this.buf.append(" ? ");
147        this.values.add(value);
148      }
149    }
150  }
151  /**
152   * 向条件类中追加包括Where语句的条件.
153   * @param column 字段。
154   * @param cond 关系。
155   * @param value 值。
156   * @since 0.0.1
157   * @since 0.0.2 IN 条件修改
158   */
159  public Condition where(String column, String cond, Object value) {
160    this.buf.append(" WHERE ");
161    this.buf.append(column);
162    this.buf.append(cond);
163    this.makeBuffer(cond, value);
164    return this;
165  }
166
167  /**
168   * 向条件类中追加AND条件.
169   * @param column 字段。
170   * @param cond 关系。
171   * @param value 值。
172   * @since 0.0.1
173   */
174  public Condition and(String column, String cond, Object value) {
175    this.buf.append(" AND ");
176    this.buf.append(column);
177    this.buf.append(cond);
178    this.makeBuffer(cond, value);
179    return this;
180  }
181
182  /**
183   * 向条件类中追加OR条件.
184   * @param column 字段。
185   * @param cond 关系。
186   * @param value 值。
187   * @since 0.0.1
188   */
189  public Condition or(String column, String cond, Object value) {
190    this.buf.append(" OR ");
191    this.buf.append(column);
192    this.buf.append(cond);
193    this.makeBuffer(cond, value);
194    return this;
195  }
196
197  /**
198   * 向条件类中追加AND条件.
199   * @param column 字段。
200   * @param cond 关系。
201   * @param value 值。
202   * @since 0.0.1
203   */
204  public Condition not(String column, String cond, Object value) {
205    this.buf.append(" NOT ");
206    this.buf.append(column);
207    this.buf.append(cond);
208    this.makeBuffer(cond, value);
209    return this;
210  }
211
212  /**
213   * 向条件类中追加子条件.
214   * @param cond 条件。
215   * @since 0.0.1
216   */
217  public void addCond(Condition cond) {
218    this.buf.append(" ( ");
219    this.buf.append(cond);
220    this.buf.append(" ) ");
221    this.values.addAll(cond.getValues());
222  }
223
224  /**
225   * 返回条件的字符串表示形式.
226   * @return String 字符串表示形式。
227   * @see java.lang.Object#toString()
228   * @since 0.0.1
229   */
230  @Override
231  public String toString() {
232    return buf.toString();
233  }
234}
