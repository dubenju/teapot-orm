001package org.teapot.db.orm;
002
003/**
004 * Torm.
005 * 因为是单实例，所以在这个类中不能存储线程的信息。
006 * 为此，引入Context的话，显着有些累赘。
007 * 做成抽象类，在pojo中的话，pojo就不单纯了。
008 * @author DBJ
009 *
010 */
011public class Torm {
012  private static volatile Torm INSTANCE;
013
014  private Torm() {
015
016  }
017
018  /**
019   * getInstance.
020   * @return Torm
021   */
022  public static Torm getInstance() {
023    if (Torm.INSTANCE == null) {
024      synchronized (Torm.class) {
025        if (Torm.INSTANCE == null) {
026          Torm.INSTANCE = new Torm();
027        }
028      }
029    }
030    return Torm.INSTANCE;
031  }
032
033  /**
034   * select.
035   * @param args String[]
036   * @return Torm
037   */
038  public static Torm select(String... args) {
039    System.out.println("SELECT");
040    for (int i = 0, size = args.length; i < size; i ++) {
041      String arg = args[i];
042      System.out.println("arg=" + arg);
043    }
044
045    return Torm.getInstance();
046  }
047
048  public Torm from() {
049    System.out.println(" FROM ");
050    return Torm.getInstance();
051  }
052}
