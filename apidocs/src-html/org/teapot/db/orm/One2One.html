017package org.teapot.db.orm;
018
019public class One2One implements IKeyValue {
020  private IKeyValue one;
021  private IKeyValue two;
022
023  public One2One() {
024      this(null, null);
025  }
026
027  public One2One(IKeyValue one1) {
028      this(one1, null);
029  }
030
031  public One2One(IKeyValue one1, IKeyValue two2) {
032    this.one = one1;
033    this.two = two2;
034  }
035
036  /**
037   * getOne.
038   * @return one IKeyValue
039   */
040  public IKeyValue getOne() {
041    return one;
042  }
043
044  /**
045   * setOne.
046   * @param one IKeyValue
047   */
048  public void setOne(IKeyValue one) {
049    this.one = one;
050  }
051
052  /**
053   * getTwo.
054   * @return two IKeyValue
055   */
056  public IKeyValue getTwo() {
057    return two;
058  }
059
060  /**
061   * setTwo.
062   * @param two IKeyValue
063   */
064  public void setTwo(IKeyValue two) {
065    this.two = two;
066  }
067
068  @Override
069  public Object get(String key) {
070    Object result = null;
071    if (this.one != null) {
072      result = this.one.get(key);
073    }
074    if (result == null) {
075      if (this.two != null) {
076        result = this.two.get(key);
077      }
078    }
079    return result;
080  }
081
082  @Override
083  public IKeyValue set(String key, Object value) {
084
085    return this;
086  }
087
088  /**
089   * @see java.lang.Object#toString()
090   */
091  @Override
092  public String toString() {
093    StringBuffer buf = new StringBuffer();
094    buf.append(this.one);
095    buf.append(" <=> ");
096    buf.append(this.two);
097    return buf.toString();
098  }
099
100}
