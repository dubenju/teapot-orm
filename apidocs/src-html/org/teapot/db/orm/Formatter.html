017package org.teapot.db.orm;
018
019/**
020 * 字符串格式化类。
021 * @author dubenju@126.com
022 * @since 0.0.1
023 */
024public enum Formatter {
025  INSTANCE;
026
027  /**
028   * 把由下划线(_)连接成的变量格式化成小驼峰式命名法（lower camel case）.
029   * 第一个单字以小写字母开始；第二个单字的首字母大写，例如：firstName、lastName。
030   * @param str 要格式化的字符串
031   * @return 格式化后的结果
032   * @since 0.0.1
033   */
034  public String fmtVar(String str) {
035    if (str == null) {
036      return str;
037    }
038    String[] aryu = str.toUpperCase().split("_");
039    String[] aryl = str.toLowerCase().split("_");
040    StringBuffer buf = new StringBuffer();
041    if (aryl.length > 0) {
042      buf.append(aryl[0]);
043      for (int idx = 1; idx < aryu.length; idx ++) {
044        if (aryu[idx].length() > 0) {
045          buf.append(aryu[idx].substring(0,  1));
046          buf.append(aryl[idx].substring(1));
047        }
048      }
049    }
050    return buf.toString();
051  }
052
053  /**
054   * 把由下划线(_)连接成的变量格式化成大驼峰式命名法（upper camel case）.
055   * 每一个单字的首字母都采用大写字母，例如：FirstName、LastName、CamelCase，也被称为Pascal命名法。
056   * @param str 要格式化的字符串
057   * @return 格式化后的结果
058   * @since 0.0.1
059   */
060  public String fmtFun(String str) {
061    if (str == null) {
062      return str;
063    }
064    String str2 = str.replaceAll("\\.", "_");
065    String[] aryu = str2.toUpperCase().split("_");
066    String[] aryl = str2.toLowerCase().split("_");
067    StringBuffer buf = new StringBuffer();
068    for (int idx = 0; idx < aryu.length; idx ++) {
069      if (aryu[idx].length() > 0) {
070        buf.append(aryu[idx].substring(0,  1));
071        buf.append(aryl[idx].substring(1));
072      }
073    }
074
075    return buf.toString();
076  }
077
078  /**
079   * 用填充位的字符串把要格式化的字符串填充成指定长度的字符串.
080   * @param in 要格式化的字符串
081   * @param ch 填充位的字符串
082   * @param len 格式化后的字符串长度
083   * @return 格式化后的结果
084   * @since 0.0.1
085   */
086  public String fmt(String in, String ch, int len) {
087    if (ch == null || ch.length() <= 0) {
088      return in;
089    }
090    int length = 0;
091    if (in != null) {
092      length = in.length();
093    }
094    int max = len - length;
095    StringBuffer buf = new StringBuffer();
096    if (max > 0) {
097      while (max > 0) {
098        buf.append(ch);
099        max --;
100      }
101    }
102    if (in != null) {
103      buf.append(in);
104    }
105    return buf.toString();
106  }
107
108  /**
109   * 把字符串格式化成大写的.
110   * @param str 要格式化的字符串
111   * @return 格式化后的结果
112   * @since 0.0.1
113   */
114  public String upper(String str) {
115    if (str == null) {
116      return str;
117    }
118    return str.toUpperCase();
119  }
120
121  /**
122   * 把字符串格式化成小写的.
123   * @param str 要格式化的字符串
124   * @return 格式化后的结果
125   * @since 0.0.1
126   */
127  public String lower(String str) {
128    if (str == null) {
129      return str;
130    }
131    return str.toLowerCase();
132  }
133}
