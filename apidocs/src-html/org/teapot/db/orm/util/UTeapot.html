017package org.teapot.db.orm.util;
018
019/**
020 * Teapot的工具类。
021 * @author dubenju@126.com
022 * @since TeapotORM 0.0.2
023 */
024public class UTeapot {
025  /**
026   * 获取路径.
027   * @param cls 类
028   * @param <T> 类型
029   * @return 路径
030   * @since TeapotORM 0.0.2
031   */
032  public static <T> String getPath(Class<T> cls) {
033    StringBuffer buf = new StringBuffer();
034    String path = cls.getResource("").getPath();
035//    System.out.println("path: " + path);
036    if (path.indexOf(".jar!") >= 0) {
037      String[] split = path.split("\\.jar!");
038//      System.out.println(split[0]);
039//      System.out.println(split[1]);
040      int posb = split[0].lastIndexOf("/");
041      int posa = split[0].indexOf("file:/");
042//      path = split[0].substring(posa + 6, posb) + split[1];
043      buf.append(split[0].substring(posa + 6, posb));
044      buf.append(split[1]);
045//      System.out.println("整形后的path：" + path);
046    } else {
047      buf.append(path);
048    }
049    return buf.toString();
050  }
051}
