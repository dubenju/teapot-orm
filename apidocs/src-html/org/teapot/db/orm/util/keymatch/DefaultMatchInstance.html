037  /**
038   * 把字符串数组转换成为Vector.
039   *
040   * @param keys 字符串数组
041   * @return Vector
042   */
043  protected static Vector<String> toVector(String[] keys) {
044    if (keys == null) {
045      return null;
046    }
047
048    Vector<String> vec = new Vector<String>();
049    for (int i = 0; i < keys.length; i++) {
050      vec.add(keys[i]);
051    }
052    return vec;
053  }

083  /**
084   * 匹配处理.
085   * @param mas Master
086   * @param tra Transaction
087   * @param mkeys 匹配
088   * @param attr 包括
089   * @param excludes 除外
090   * @param proc 匹配处理

095  @SuppressWarnings("unchecked")
096  public <T1 extends IKeyValue, T2 extends IKeyValue> void keyMatch(List<T1> mas,
097      List<T2> tra, String[] mkeys,
098      String[] attr, String[] excludes, IMatch proc, Object arg)
099      throws Exception {
100    this.loMas = (List<IKeyValue>) mas;
101    this.loTra = (List<IKeyValue>) tra;
102    this.mkeys = toVector(mkeys);
103    setAttr(toVector(attr));
104    this.arg = arg;
105
106    // 执行 主处理方法.
107    run(proc);
108  }

110  /*
111   * 这种方法与上面的主要逻辑run（）方法有很大的关系。
112   * 所以这个方法只是一个主要逻辑的子程序。
113   * 小心！ 不要以类别成员方法的方式来思考这个方法。
114   * 所以，这个方法设置为“final”。
115   */
116  /**
117   * 匹配处理.
118   * @param mas Master
119   * @param tra Transaction
120   * @param proc 匹配处理
121   * @param arg 参数
122   * @throws Exception 处理过程中发生的异常。
123   * @since TeapotORM 0.0.2
124   */
125  private final void runKeyMatch(IKeyValue mas, IKeyValue tra, IMatch proc, Object arg)
126      throws Exception {
127    int compAttr;
128    boolean blAttr = false;
129
130    for (int i = 1; ; i ++) {
131      // 比较属性部
132      compAttr = comAttrs(mas, tra, this.att);
133      ATTR: switch (compAttr) {
134        case MATCHED:
135          blAttr = true;
136          break ATTR;
137        default:
138          blAttr = false;
139          break ATTR;
140        }
141
142      if (blAttr == false) {
143        proc.mismatch(mas, tra, arg);
144      }
145
146      // 只有属性部也匹配的时候才执行doMatch().
147      if (blAttr == true && proc.match(mas, tra, arg)) {
148        return;
149      }
150
151      // 如果matching返回true继续。
152      mas = (IKeyValue) getMatchingMasterAhead(i);
153      if (mas == null) {
154        return;
155      }
156    }
157  }
158
159  /**
160   * 主处理.
161   * @param proc 回调
162   * @param <T1> 数据类型.
163   * @param <T2> 数据类型.
164   * @throws Exception 处理过程中发生的异常.
165   */
166  protected <T1 extends IKeyValue, T2 extends IKeyValue> void run(IMatch proc) throws Exception {
167    // 读取Master
168    IKeyValue mas = readMas();
169    // 读取Transaction
170    IKeyValue tra = readTra();
171
172    if (!NullList.isNull(mas, tra, proc, this)) {
173      return;
174    }
175
176    boolean isMatched = false; // Generally speacking this flag is for Tra
177    boolean isMasterUsed = false; // This flag is for Mas
178    IKeyValue masMatched = null; // for backup 1(tra):N(mas) matching
179    int comp;
180
181    MAIN: for (;;) {
182      // 循环transaction
183      COMP: for (;;) {
184        // 循环master
185        if (mas == null) {
186          // if master cannot be read, matching impossible.
187          break COMP;
188        }
189
190        // 比较 keys
191        comp = compareKeys(mas, tra, mkeys);
192        switch (comp) {
193          case MATCHED:
194            isMasterUsed = true; // don't care about the result of
195            isMatched = true; // doMatching() => Deterministic
196            masMatched = mas;
197            // matching
198            // attributeMismatch
199            runKeyMatch(mas, tra, proc, arg);
200            break COMP;
201
202          case TRA_AHEAD:
203            // if never used (if keys never matched with any master)
204            if (!isMasterUsed) {
205              proc.masterOnly(mas, arg);
206            } else {
207              masMatched = mas;
208            }
209            mas = readMas();
210            isMasterUsed = transactionAhead(mas, masMatched, mkeys, isMasterUsed);
211            continue COMP;
212
213          case MAS_AHEAD:
214            break COMP;
215
216          default:
217            return;
218        }
219      } // COMP
220
221      if (!isMatched) {
222        // if no matching record exists, do some proc
223        proc.transactionOnly(tra, arg);
224      }
225
226      // read next transaction and initialize the control flag
227      tra = readTra();
228      if (tra == null) {
229        break MAIN;
230      }
231
232      isMatched = false;
233    } // MAIN
234
235    finish(mas, masMatched, mkeys, proc, arg);
236  }

238  /**
239   * This method is additionally introduced for 1(Tra):N(Mas) mathcing. Expected
240   * Use) This method is expected to be called from doMatch(). This method is
241   * necessary to check whether the following records might have the same
242   * matching keys. Policy of Matching Procedure Frame) I. The index of master
243   * and transaction never decrement. II. The index of master and transaction
244   * never be controlled from outside of this class (private state). No control
245   * method is provided. Case Study) 1. Keys are matched but attributes such as
246   * effective data are invalid. 2. The following master record(s) might have
247   * the same keys and the valid attributes. 3. Use the valid master record as
248   * the matching AND "valid" record. 4. The next "transaction" may have the
249   * same keys. 5. The next transaction may find matching and valid record among
250   * master records, which are already examined with the previous transaction.
251   * Note: This method can be accessed from non-children classes if they are in
252   * the same package. But introducing sub packages requires to let this be
253   * public.
254   * @param offset offset
255   * @return Object 结果
256   */
257  public Object getMatchingMasterAhead(int offset) {
258    int current = getMasterIndex();
259    if (this.loMas == null || this.loMas.size() == 0 || offset < 0
260        || this.loMas.size() <= current + offset) {
261      return null;
262    }
263
264    // since idx was moved forward by one after read()
265    if (compareMasterKeys(current, offset)) {
266      return this.loMas.get(current + offset);
267    } else {
268      return null;
269    }
270  }
271
272  /**
273   * Don't leak information (Keep private/encupsulation/information hiding) =>
274   * Outside of this class, no information about index, current/offset.
275   */
276  private boolean compareMasterKeys(int current, int offset) {
277    IKeyValue currentRecord = (IKeyValue) getMaster(current);
278    IKeyValue aheadRecord   = (IKeyValue) getMaster(current + offset);
279    if (aheadRecord == null) {
280      return false;
281    }
282    for (int i = 0; i < this.mkeys.size(); i++) {
283      @SuppressWarnings({ "unchecked", "rawtypes" })
284      int comp = ((Comparable) currentRecord.get((String) this.mkeys.get(i)))
285          .compareTo(aheadRecord.get((String) this.mkeys.get(i)));
286      if (comp > 0) {
287        return false;
288      } else if (comp < 0) {
289        return false;
290      }
291    } // COMP
292    return true;
293  }

295  /**
296   * Don't leak information (Keep private/encupsulation/information hiding) =>
297   * Outside of this class, no information about index, current/offset.
298   */
299  private Object getMaster(int position) {
300    if (this.loMas == null || this.loMas.size() == 0
301        || this.loMas.size() <= position) {
302      return null;
303    }
304    return this.loMas.get(position); // since idx was moved forward by one after read()
305  }

307  /**
308   * This method is heavily related with the above main logic, run() method.
309   * Therefore this method is just a subrutione of the main logic. Caution ! DO
310   * NOT THINK ABOUT THIS METHOD AS A CLASS MEMBER METHOD. So, this method is
311   * set to "final".
312   */
313  private final boolean transactionAhead(IKeyValue tra, IKeyValue traMatched,
314      List<String> mkeys, boolean isMasterUsed) {
315    int compTra;
316    if (tra == null) {
317      return isMasterUsed;
318    }
319    // for the key comparison with the following master record
320    if (traMatched == null) {
321      compTra = MAS_AHEAD;
322    } else {
323      compTra = compareKeys(tra, traMatched, mkeys);
324    }
325    TRA_KEY: switch (compTra) {
326      case MATCHED:
327        isMasterUsed = true;
328        break TRA_KEY;
329      default:
330        isMasterUsed = false;
331        break TRA_KEY;
332    }
333    return isMasterUsed;
334  } // transactionAhead
335
336  /**
337   * This method is heavily related with the above main logic, run() method.
338   * Therefore this method is just a subrutione of the main logic. Caution ! DO
339   * NOT THINK ABOUT THIS METHOD AS A CLASS MEMBER METHOD. So, this method is
340   * set to "final".
341   */
342  private final void finish(IKeyValue mas, IKeyValue masMatched, List<String> mkeys,
343      IMatch proc, Object param) throws Exception {
344    // Calling back procedure class to let it know the no transaction
345    // won't be examined for the lest of master records
346    if (mas == null) {
347      return;
348    }
349
350    int compMas;
351    boolean isMasterUsed;
352    while (true) {
353      // for the key comparison with the following
354      // master records
355      if (masMatched == null) {
356        compMas = MAS_AHEAD;
357      } else {
358        compMas = compareKeys(mas, masMatched, mkeys);
359      }
360      MAS_KEY: switch (compMas) {
361        case MATCHED:
362          isMasterUsed = true;
363          break MAS_KEY;
364        default:
365          isMasterUsed = false;
366          masMatched = null; // Once un-matched, Never match again
367          // Unless null specified, meaningless
368          // key comparision will be run
369          break MAS_KEY;
370      }
371
372      if (!isMasterUsed) {
373        proc.masterOnly(mas, param);
374      }
375
376      mas = readMas();
377      if (mas == null) {
378        return;
379      }
380    }
381  }

383  @SuppressWarnings("unchecked")
384  protected int compareKeys(IKeyValue mas, IKeyValue tra, List<String> mkeys) {
385    for (int i = 0; i < mkeys.size(); i++) {
386      String mkey = (String) mkeys.get(i);
387      int comp = 0;
388      try {
389        System.out.println("[DefaultMatchInstance::compareKeys]" + mas + "=" + tra + "=" + mkey);
390        @SuppressWarnings("rawtypes")
391        Comparable c1 = (Comparable) tra.get(mkey);
392        @SuppressWarnings("rawtypes")
393        Comparable c2 = (Comparable) mas.get(mkey);
394        comp = c1.compareTo(c2);
395      } catch (ClassCastException cce) {
396        cce.printStackTrace();
397      }
398
399      if (comp > 0) {
400        return TRA_AHEAD;
401      } else if (comp < 0) {
402        return MAS_AHEAD;
403      }
404    } // COMP
405    return 0;
406  }

408  /**
409   * 比较属性.
410   * @param mas Master
411   * @param tra Transaction
412   * @param attr 属性
413   * @return 比较结果。
414   */
415  protected int comAttrs(IKeyValue mas, IKeyValue tra, List<String> attr) {
416    if (attr == null) {
417      // System.out.println("[Warning!] Keys for Attributes is null!");
418      return 0;
419    }
420    COMP: for (int i = 0; i < attr.size(); i++) {
421      // Get attribute name
422      String name = (String) attr.get(i);
423
424      // Retrieve object as object to check type
425      @SuppressWarnings("rawtypes")
426      Comparable traObj = (Comparable) tra.get(name);
427      @SuppressWarnings("rawtypes")
428      Comparable masObj = (Comparable) mas.get(name);
429
430      if (traObj == null && masObj == null) {
431        continue COMP; // handle the same attribute ( null == null )
432      }
433      if (traObj == null) {
434        return MAS_AHEAD;
435      }
436      if (masObj == null) {
437        return TRA_AHEAD;
438      }
439
440      @SuppressWarnings("unchecked")
441      int comp = traObj.compareTo(masObj);
442
443      if (comp != 0 && traObj instanceof String) {
444        String enc1 = (String) traObj;
445        String enc2 = (String) masObj;
446
447        comp = enc1.compareTo(enc2);
448        if (comp == 0) {
449          System.out.println("[MatchingFrameAttribute] Char Code Different"
450              + traObj + "/" + masObj);
451        }
452      }
453
454      if (comp > 0) {
455        return TRA_AHEAD;
456      } else if (comp < 0) {
457        return MAS_AHEAD;
458      }
459    } // COMP
460    return 0;
461  }

463  protected IKeyValue readMas() {
464    if (this.loMas == null || this.loMas.size() == 0
465        || this.loMas.size() == this.idxMas) {
466      return null;
467    }
468    return (IKeyValue) this.loMas.get(this.idxMas++);
469  }
470
471  protected IKeyValue readTra() {
472    if (this.loTra == null || this.loTra.size() == 0
473        || this.loTra.size() == this.idxTra) {
474      return null;
475    }
476    return (IKeyValue) this.loTra.get(this.idxTra++);
477  }

479  /**
480   * 返回Master索引.
481   * @return Master索引.
482   */
483  public int getMasterIndex() {
484    return this.idxMas - 1;
485  }
