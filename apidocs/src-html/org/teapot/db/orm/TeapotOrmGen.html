184  /**
185   * 通过XML读取表信息.
186   * @param fileName XML
187   * @return 表信息
188   */
189  public static List<MdlTable> readXmlFile(String fileName) {
190    DocumentBuilderFactory builderFactory = DocumentBuilderFactory.newInstance();
191    Document document = null;
192    try {
193      //DOM parser instance
194      DocumentBuilder builder = builderFactory.newDocumentBuilder();
195      //parse an XML file into a DOM tree
196      document = builder.parse(new File(fileName));
197    } catch (ParserConfigurationException e) {
198      e.printStackTrace();
199    } catch (SAXException e) {
200      e.printStackTrace();
201    } catch (IOException e) {
202      e.printStackTrace();
203    }
204
205    //get root element
206    Element rootElement = document.getDocumentElement();
207
208    List<MdlTable> result = new ArrayList<MdlTable>();
209    NodeList nodeList = rootElement.getElementsByTagName("table");
210    if (nodeList != null) {
211      for (int i = 0 ; i < nodeList.getLength(); i ++)  {
212        Element element = (Element)nodeList.item(i);
213        String name = element.getAttribute("name");
214        String type = element.getAttribute("type");
215        System.out.println(name + "," + type);
216        MdlTable mdlTable = new MdlTable();
217        mdlTable.setTableName(name.toUpperCase());
218        ArrayList<MdlColumn> colmuns = new ArrayList<MdlColumn>();
219        NodeList subNodeList = element.getElementsByTagName("column");
220        if (subNodeList != null) {
221          for (int j = 0 ; j < subNodeList.getLength(); j ++)  {
222            Element elem = (Element)subNodeList.item(j);
223            String columnName = elem.getAttribute("name");
224            String typeName = elem.getAttribute("type");
225            System.out.println(columnName + "," + typeName);
226            MdlColumn col = new MdlColumn();
227            col.setColumnName(columnName);
228            col.setTypeName(typeName);
229            col.setTypeNameJava(
230                TypeJavaDb.getTypeJava(TeapotDb.getInstance().getDbType()).getJavaType(typeName));
231            colmuns.add(col);
232          }
233        }
234        mdlTable.setColmuns(colmuns);
235        result.add(mdlTable);
236      }
237    }
238    return result;
239  }
